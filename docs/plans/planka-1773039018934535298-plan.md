```typescript
export async function getSessionById(id: string): Promise<GameSessionState | null> {
  const dbSession = await prisma.gameSession.findUnique({
    where: { id },
    select: { pin: true },
  })
  if (!dbSession) return null
  return getSessionByPin(dbSession.pin)
}
```
